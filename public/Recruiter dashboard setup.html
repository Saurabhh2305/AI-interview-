tsx
"use client";

import { useState, useEffect } from "react";
import { useRouter } from "next/navigation";
import axios from "axios";
import {
  Card,
  CardHeader,
  CardTitle,
  CardDescription,
  CardContent,
} from "@/components/ui/card";
import { Button } from "@/components/ui/button";
import { Input } from "@/components/ui/input";
import { Label } from "@/components/ui/label";
import { Textarea } from "@/components/ui/textarea";
import {
  LayoutDashboard,
  Briefcase,
  FileBarChart,
  Settings,
  LogOut,
  UserCheck,
  Plus,
} from "lucide-react";
import {
  Dialog,
  DialogContent,
  DialogHeader,
  DialogTitle,
  DialogFooter,
} from "@/components/ui/dialog";

const BASE_URL = "http://localhost:8080/api/jobs";

export default function RecruiterDashboard() {
  const router = useRouter();
  const [recruiterName] = useState<string>("Saurabh Gupta");
  const [open, setOpen] = useState(false);
  const [loading, setLoading] = useState(false);
  const [jobs, setJobs] = useState<any[]>([]);
  const [jobData, setJobData] = useState({
    title: "",
    description: "",
    skillsRequired: "",
    location: "",
    experienceLevel: "",
    salaryRange: "",
  });

  const token =
    typeof window !== "undefined" ? localStorage.getItem("token") : null;

  //  Fetch all jobs
  const fetchJobs = async () => {
    try {
      const response = await axios.get(`${BASE_URL}/1`, {
        headers: {
          Authorization: `Bearer ${token}`,
          Accept: "application/json",
        },
      });
      if (response.status === 200) {
        setJobs(response.data.jobs || []); // Adjust according to backend structure
      }
    } catch (error) {
      console.error("Error fetching jobs:", error);
    }
  };

  useEffect(() => {
    fetchJobs();
  }, []);

  //  Logout
  const handleLogout = () => {
    localStorage.removeItem("token");
    router.push("/auth/login");
  };

  //  Handle form field change
  const handleChange = (
    e: React.ChangeEvent<HTMLInputElement | HTMLTextAreaElement>
  ) => {
    setJobData({ ...jobData, [e.target.name]: e.target.value });
  };

  //  Submit Job to Backend
  const handleSubmit = async () => {
    try {
      setLoading(true);
      const payload = { ...jobData, recruiterId: 1 };
      const response = await axios.post(`${BASE_URL}/create`, payload, {
        headers: {
          Authorization: `Bearer ${token}`,
          "Content-Type": "application/json",
        },
      });

      if (response.status === 200 || response.status === 201) {
        alert(" Job created successfully!");
        setJobData({
          title: "",
          description: "",
          skillsRequired: "",
          location: "",
          experienceLevel: "",
          salaryRange: "",
        });
        setOpen(false);
        fetchJobs(); // refresh job list
      } else {
        alert(" Failed to create job. Please try again.");
      }
    } catch (error) {
      console.error("Error creating job:", error);
      alert(" Error while creating job. Check console for details.");
    } finally {
      setLoading(false);
    }
  };

  return (
    <div className="min-h-screen flex bg-gradient-to-br from-blue-50 to-blue-100">
      {/* Sidebar */}
      <aside className="hidden md:flex flex-col w-64 bg-white shadow-md border-r border-gray-200">
        <div className="p-6 border-b">
          <h2 className="text-2xl font-bold text-blue-600 tracking-tight">
            Recruiter Panel
          </h2>
          <p className="text-sm text-gray-500 mt-1">Welcome, {recruiterName}</p>
        </div>

        <nav className="flex-1 p-4 space-y-2">
          <Button
            variant="ghost"
            className="w-full justify-start text-gray-700 hover:bg-blue-50"
            onClick={() => router.push("/dashboard/recruiter")}
          >
            <LayoutDashboard className="mr-3 h-5 w-5 text-blue-500" />
            Dashboard Overview
          </Button>

          <Button
            variant="ghost"
            className="w-full justify-start text-gray-700 hover:bg-blue-50"
            onClick={() => router.push("/dashboard/recruiter/jobs")}
          >
            <Briefcase className="mr-3 h-5 w-5 text-blue-500" />
            Manage Job Posts
          </Button>

          <Button
            variant="ghost"
            className="w-full justify-start text-gray-700 hover:bg-blue-50"
            onClick={() => router.push("/dashboard/recruiter/applicants")}
          >
            <UserCheck className="mr-3 h-5 w-5 text-blue-500" />
            View Applicants
          </Button>

          <Button
            variant="ghost"
            className="w-full justify-start text-gray-700 hover:bg-blue-50"
            onClick={() => router.push("/dashboard/recruiter/reports")}
          >
            <FileBarChart className="mr-3 h-5 w-5 text-blue-500" />
            Reports & Analytics
          </Button>

          <Button
            variant="ghost"
            className="w-full justify-start text-gray-700 hover:bg-blue-50"
            onClick={() => router.push("/dashboard/recruiter/settings")}
          >
            <Settings className="mr-3 h-5 w-5 text-blue-500" />
            Settings
          </Button>
        </nav>

        <div className="p-4 border-t">
          <Button
            variant="destructive"
            className="w-full flex items-center justify-center"
            onClick={handleLogout}
          >
            <LogOut className="mr-2 h-5 w-5" /> Logout
          </Button>
        </div>
      </aside>

      {/* Main Content */}
      <main className="flex-1 p-6 md:p-10 space-y-8 overflow-y-auto">
        <div className="flex flex-col md:flex-row md:items-center md:justify-between">
          <h1 className="text-3xl font-bold text-gray-800">
            Recruiter Dashboard 
          </h1>
          <Button
            onClick={() => setOpen(true)}
            className="bg-blue-600 hover:bg-blue-700 text-white mt-4 md:mt-0 flex items-center"
          >
            <Plus className="mr-2 h-5 w-5" />
            Create New Job
          </Button>
        </div>

        {/*  Job List Section */}
        <section>
          <h2 className="text-2xl font-semibold mb-4">Your Job Posts</h2>
          {jobs.length === 0 ? (
            <p className="text-gray-500">No job posts yet.</p>
          ) : (
            <div className="grid grid-cols-1 md:grid-cols-2 lg:grid-cols-3 gap-6">
              {jobs.map((job, index) => (
                <Card
                  key={index}
                  className="hover:shadow-lg transition-all duration-200 border border-gray-200"
                >
                  <CardHeader>
                    <CardTitle className="text-blue-700">{job.title}</CardTitle>
                    <CardDescription>{job.location}</CardDescription>
                  </CardHeader>
                  <CardContent>
                    <p className="text-gray-700 text-sm mb-2">
                      {job.description}
                    </p>
                    <p className="text-sm text-gray-600">
                      <strong>Skills:</strong> {job.skillsRequired}
                    </p>
                    <p className="text-sm text-gray-600">
                      <strong>Experience:</strong> {job.experienceLevel}
                    </p>
                    <p className="text-sm text-gray-600">
                      <strong>Salary:</strong> {job.salaryRange}
                    </p>
                  </CardContent>
                </Card>
              ))}
            </div>
          )}
        </section>
      </main>

      {/*  Create Job Dialog */}
      <Dialog open={open} onOpenChange={setOpen}>
        <DialogContent className="sm:max-w-lg">
          <DialogHeader>
            <DialogTitle>Create a New Job Post</DialogTitle>
          </DialogHeader>
          <div className="space-y-4">
            <div>
              <Label>Job Title</Label>
              <Input
                name="title"
                value={jobData.title}
                onChange={handleChange}
                placeholder="e.g. Senior Java Developer"
              />
            </div>
            <div>
              <Label>Description</Label>
              <Textarea
                name="description"
                value={jobData.description}
                onChange={handleChange}
                placeholder="Brief job description..."
              />
            </div>
            <div>
              <Label>Skills Required</Label>
              <Input
                name="skillsRequired"
                value={jobData.skillsRequired}
                onChange={handleChange}
                placeholder="Java, Spring Boot, AWS"
              />
            </div>
            <div>
              <Label>Location</Label>
              <Input
                name="location"
                value={jobData.location}
                onChange={handleChange}
                placeholder="Bangalore"
              />
            </div>
            <div>
              <Label>Experience Level</Label>
              <Input
                name="experienceLevel"
                value={jobData.experienceLevel}
                onChange={handleChange}
                placeholder="5-8 years"
              />
            </div>
            <div>
              <Label>Salary Range</Label>
              <Input
                name="salaryRange"
                value={jobData.salaryRange}
                onChange={handleChange}
                placeholder="15-20 LPA"
              />
            </div>
          </div>

          <DialogFooter className="mt-4">
            <Button
              variant="outline"
              onClick={() => setOpen(false)}
              disabled={loading}
            >
              Cancel
            </Button>
            <Button
              onClick={handleSubmit}
              className="bg-blue-600 text-white hover:bg-blue-700"
              disabled={loading}
            >
              {loading ? "Submitting..." : "Create Job"}
            </Button>
          </DialogFooter>
        </DialogContent>
      </Dialog>
    </div>
  );
}
